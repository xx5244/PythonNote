    ```python
    import matplotlib.pyplot as plt
    import numpy as np

    x = np.linspace(0.0, 5.0)

    y1 = np.sin(x)
    y2 = np.cos(x)

    # 四條繪製在同張圖上
    # plt.plot(x, y1, 'b--')
    # plt.plot(x, y2, 'r--')
    # plt.plot(x, y1, 'r*')
    # plt.plot(x, y2, 'b*')

    # 四張用子圖形式
    # plt.subplot(2, 2, 1)
    # plt.plot(x, y1, 'b--')
    # plt.subplot(2, 2, 2)
    # plt.plot(x, y2, 'r--')
    # plt.subplot(2, 2, 3)
    # plt.plot(x, y1, 'r*')
    # plt.subplot(2, 2, 4)
    # plt.plot(x, y2, 'b*')

    # plt.show()

    """
    subplot(*args)
    :*args: int，default:(1, 1, 1) -> (numRows, numCols, plotNum)
    :numRows: int，row的數量
    :numCols: int，column的數量
    :plotNum: int，要畫圖的編號
    總格子數為row數 * columns數，編號是從左上到右下的
    例如:(2, 2, 1)即將畫作分成4格，2row * 2coulumn當中的編號1，如上面的範例
    每次的圖都是等同重新分割一樣，然後一層一層疊上去的感覺，如下面的範例
    211 -> 將畫作分成2格，2row * 1col，如圖1，接著作畫在編號1的位置
    234 -> 將畫作分成6格，2row * 3col，如圖2，接著作畫在編號4的位置
    注意:最好是把每次分割都當成獨立個體，最後在疊合起來，也就是說，如果第一次是211，而第2次是231的畫，由於位置重複，前面的就會被後面的作畫給覆蓋過去
    """

    # 範例

    # plt.subplot(2, 3, 4)
    # plt.subplot(2, 3, 5)
    ```
